    <script>
        // BASE DE DATOS LOCAL
        const pilotos = { 
            "101": { n: "Carlos Hernandez", t: "Inter transportes", c: "Agribrands Purina Guatemala S.A", p: "C-834BPT" }, 
            "102": { n: "Mynor Josue Guzman Rivera", t: "Inter transportes", c: "Agribrands Purina Guatemala S.A", p: "C-826BZQ" } 
        };

        // REEMPLAZA ESTO CON TU URL DE GOOGLE APPS SCRIPT
        const URL_SCRIPT = "TU_URL_DE_GOOGLE_AQUI"; 

        function entrar() {
            const user = document.getElementById('u').value.toLowerCase().trim();
            const pass = document.getElementById('p').value;

            if(user === "operador 1" && pass === "terpac1") {
                document.getElementById('login').style.display='none'; 
                document.getElementById('ui-app').style.display='block';
            } 
            else if(user === "supervisor 1" && pass === "terpac2") {
                document.getElementById('login').style.display='none'; 
                document.getElementById('ui-supervisor').style.display='block';
                iniciarReloj();
                actualizarPildoras();
            }
            else { alert("Acceso Incorrecto"); }
        }

        function check() {
            const id = document.getElementById('id-in').value.trim();
            const box = document.getElementById('piloto-box');
            if(pilotos[id]) { 
                box.innerText = "👨‍✈️ Piloto: " + pilotos[id].n; 
                box.style.display='block'; 
            } else { 
                box.style.display='none'; 
            }
        }

        function iniciarReloj() {
            setInterval(() => {
                const ahora = new Date();
                document.getElementById('reloj').innerText = ahora.toLocaleTimeString();
                document.getElementById('fecha-actual').innerText = ahora.toLocaleDateString();
            }, 1000);
        }

        async function actualizarPildoras() {
            if(URL_SCRIPT === "TU_URL_DE_GOOGLE_AQUI") return;
            try {
                const response = await fetch(URL_SCRIPT);
                const data = await response.json();
                document.getElementById('count-pendientes').innerText = data.pendientes.toString().padStart(2, '0');
                document.getElementById('count-proceso').innerText = data.proceso.toString().padStart(2, '0');
                document.getElementById('count-despachadas').innerText = data.despachadas.toString().padStart(2, '0');
            } catch (e) { console.log("Error cargando pildoras"); }
        }

        function enviar() {
            const id = document.getElementById('id-in').value.trim();
            if(!pilotos[id]) return alert("ID no encontrado");
            
            const btn = document.getElementById('btn-env');
            btn.innerText = "Enviando..."; 
            btn.disabled = true;

            const datos = {
                id: id,
                piloto: pilotos[id].n,
                mov: document.getElementById('mov').value,
                prod: document.getElementById('prod').value,
                transp: pilotos[id].t,
                cliente: pilotos[id].c,
                placa: pilotos[id].p,
                cTer: document.getElementById('CicloTer').value,
                cEPQ: document.getElementById('CicloEPQ').value
            };

            // Convertir a formato que Google Sheets entiende fácilmente
            const cuerpo = new URLSearchParams(datos);

            fetch(URL_SCRIPT, {
                method: 'POST',
                mode: 'no-cors',
                body: cuerpo
            }).then(() => {
                alert("✅ Registro Exitoso");
                document.getElementById('id-in').value = "";
                document.getElementById('CicloTer').value = "";
                document.getElementById('CicloEPQ').value = "";
                document.getElementById('piloto-box').style.display='none';
            }).catch(() => {
                alert("Error de envío");
            }).finally(() => {
                btn.innerText = "REGISTRAR DATOS 🚀";
                btn.disabled = false;
            });
        }
    </script>
